  4. set up ticks, labels and legends

## Set up figure object and sub-plots and plotting on each sub-plot should be done in the same cell to dispaly all the graph together.
## Set up figure object
fig = plt.figure(figsize=(10,10)) ## change figure size

## Set up sub plot of the figure
subplot1 = fig.add_subplot(2,2,1) ## nrows, ncols, index
subplot2 = fig.add_subplot(2,2,2)
subplot3 = fig.add_subplot(2,2,4)

## First Sub-plot
df_vs2_sub = df_vs2[df_vs2['USUBJID']=='01-701-1015']
subplot1.plot(df_vs2_sub['VISITNUM'], df_vs2_sub['VSSTRESN'], linestyle='--', color='r', marker = 'o') #subplot2 - plot
subplot1.set_title('Vital Sign for subject of 01-701-1015')  #subplot2 - title
subplot1.set_xlabel('Visit Num')                             #subplot2 - label x
subplot1.set_ylabel('Blood Pressure (mmHg)')                 #subplot2 - label y

## Second Sub-plot
df_vs2_sub = df_vs2[df_vs2['USUBJID']=='01-701-1028']
subplot2.plot(df_vs2_sub['VISITNUM'], df_vs2_sub['VSSTRESN'], linestyle='--', color='r', marker = 'o')
subplot2.set_title('Vital Sign for subject of 01-701-1028')
subplot2.set_xlabel('Visit Num')
subplot2.set_ylabel('Blood Pressure (mmHg)')

## Third sub-plot
df_vs2_sub = df_vs2[df_vs2['USUBJID']=='01-718-1355']
